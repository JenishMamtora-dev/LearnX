Advanced C++ Topics

1. Exception Handling

Exception handling allows graceful handling of runtime errors using try, catch, throw.

Example

#include <iostream>
using namespace std;

int main() {
    int a=10,b=0;
    try {
        if(b==0) throw "Division by zero!";
        cout << a/b;
    } catch(const char* msg) {
        cout << "Error: " << msg;
    }
}
Error: Division by zero!

2. Namespaces

Namespaces prevent naming conflicts in large programs.

Example

#include <iostream>
namespace A { int x=5; }
namespace B { int x=10; }
using namespace std;
int main() {
    cout << A::x << " " << B::x;
}
5 10

3. Lambda Functions

Lambda functions are anonymous functions used for short tasks.

Example

#include <iostream>
using namespace std;

int main() {
    auto add = [](int a,int b){ return a+b; };
    cout << add(5,3);
}
8

4. Smart Pointers (C++11)

#include <iostream>
#include <memory>
using namespace std;

int main() {
    unique_ptr<int> p = make_unique<int>(100);
    cout << *p;
}
100

5. Multithreading (C++11)

#include <iostream>
#include <thread>
using namespace std;

void hello() { cout << "Hello from thread"; }

int main() {
    thread t1(hello);
    t1.join();
}
Hello from thread
